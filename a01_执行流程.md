```bash

# 第一次提交
git init
git add . 
git commit -m "first commit" # 提交更改
git branch -M main # 创建main分支
git remote add origin https://github.com/erthorpabar/github-docker-CICD.git # 推送位置
git push -u origin main # 推送

# 后续提交
git add .
git commit -m "xxx"
git push origin main

# 创建 README.md（如果还没有的话）
echo "# github-docker-CICD" >> README.md

# 初始化 git 仓库
git init

# 添加所有文件
git add .

# 提交更改
git commit -m "first commit"

# 创建 main 分支
git branch -M main

# 添加远程仓库（如果还没添加）
git remote add origin https://github.com/erthorpabar/github-docker-CICD.git

# 推送
git push origin main

# 第一次推送
git push -u origin main

# 删除git历史记录
Remove-Item -Recurse -Force .git

```
